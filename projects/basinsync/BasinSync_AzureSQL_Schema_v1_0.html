CREATE TABLE dbo.vendors (
    vendor_id           INT           IDENTITY(1,1)  NOT NULL,
    vendor_code         CHAR(3)                        NOT NULL,
    registered_name     NVARCHAR(200)                  NOT NULL,
    abn                 CHAR(11)                       NOT NULL,
    azure_ad_vendor_id  NVARCHAR(100),
    tier                TINYINT                        NOT NULL,
    gst_registered     BIT           DEFAULT 1        NOT NULL,
    status             VARCHAR(20)   DEFAULT 'ACTIVE' NOT NULL,
    CONSTRAINT PK_vendors         PRIMARY KEY (vendor_id),
    CONSTRAINT UQ_vendor_code     UNIQUE      (vendor_code),
    CONSTRAINT UQ_vendor_abn      UNIQUE      (abn),
    CONSTRAINT CK_vendor_tier     CHECK       (tier IN (1,2,3)),
    CONSTRAINT CK_vendor_status   CHECK       (status IN ('ACTIVE','SUSPENDED','MONITOR','INACTIVE'))
);

CREATE TABLE dbo.projects (
    project_wbs         VARCHAR(30)   NOT NULL,
    project_name        NVARCHAR(300) NOT NULL,
    utility_name        NVARCHAR(100) NOT NULL,
    contract_value      DECIMAL(14,2) NOT NULL,
    monthly_cap         DECIMAL(14,2),
    regulator           VARCHAR(20)   NOT NULL,
    sopa_window_days    TINYINT       DEFAULT 10 NOT NULL,
    pm_routing_key      VARCHAR(50)   NOT NULL,
    capex_opex_profile  VARCHAR(20)   NOT NULL,
    CONSTRAINT PK_projects          PRIMARY KEY (project_wbs),
    CONSTRAINT CK_projects_regulator CHECK       (regulator IN ('IPART_NSW','ESC_VIC','ESCOSA_SA')),
    CONSTRAINT CK_projects_capex     CHECK       (capex_opex_profile IN ('CAP_HEAVY','OPX_HEAVY','MIXED'))
);

CREATE TABLE dbo.claim_headers (
    claim_id             VARCHAR(30)     NOT NULL,    -- GS-YYYY-MM-VND-SEQ
    vendor_id            INT             NOT NULL,
    project_wbs          VARCHAR(30)     NOT NULL,
    claim_period         CHAR(7)         NOT NULL,    -- YYYY-MM
    submission_timestamp DATETIME2(0)    NOT NULL,
    sopa_deadline        DATE            NOT NULL,    -- immutable after write
    claim_status         VARCHAR(20)     NOT NULL,
    total_ex_gst         DECIMAL(14,2)   NOT NULL,
    total_gst            DECIMAL(14,2)   NOT NULL,
    total_inc_gst        DECIMAL(14,2)   NOT NULL,
    pm_id                INT,
    azure_validation_id  VARCHAR(30),                -- AZF-VAL-YYYYMMDD-XXXXX
    advisory_flags       NVARCHAR(MAX),              -- JSON array of rule advisories
    blob_storage_url     NVARCHAR(500),
    created_at           DATETIME2(0)    DEFAULT GETUTCDATE(),
    CONSTRAINT PK_claim_headers    PRIMARY KEY (claim_id),
    CONSTRAINT CK_claim_status     CHECK       (claim_status IN (
        'RECEIVED','FN1_RUNNING','FN1_BLOCKED','FN2_RUNNING',
        'FN2_FAILED','PM_REVIEW','PM_QUERIED','APPROVED',
        'REJECTED','PAID','ESCALATED')),
    CONSTRAINT FK_claims_vendor    FOREIGN KEY (vendor_id)   REFERENCES dbo.vendors  (vendor_id),
    CONSTRAINT FK_claims_project   FOREIGN KEY (project_wbs) REFERENCES dbo.projects (project_wbs)
);

CREATE TABLE dbo.claim_lines (
    line_id              INT           IDENTITY(1,1)  NOT NULL,
    claim_id             VARCHAR(30)                    NOT NULL,
    line_sequence        SMALLINT                       NOT NULL,
    wbs_element          VARCHAR(50)                    NOT NULL,
    sap_cost_element     VARCHAR(10),
    capex_opex_flag      CHAR(3)                        NOT NULL,
    description          NVARCHAR(500)                  NOT NULL,
    uom                  VARCHAR(10)                    NOT NULL,
    contract_qty         DECIMAL(14,3),
    contract_rate        DECIMAL(14,2),
    prev_claimed_pct     DECIMAL(5,4)                   NOT NULL,
    this_claim_pct       DECIMAL(5,4)                   NOT NULL,
    this_claim_amt_ex    DECIMAL(14,2)                  NOT NULL,
    gst_amt              DECIMAL(14,2)                  NOT NULL,
    this_claim_amt_inc   DECIMAL(14,2)                  NOT NULL,
    invoice_ref          NVARCHAR(100),
    rate_anomaly_flag    BIT           DEFAULT 0        NOT NULL,
    CONSTRAINT PK_claim_lines       PRIMARY KEY (line_id),
    CONSTRAINT UQ_claim_line_seq    UNIQUE      (claim_id, line_sequence),
    CONSTRAINT CK_capex_opex        CHECK       (capex_opex_flag IN ('CAP','OPX')),
    CONSTRAINT CK_gst_calc          CHECK       (ABS(gst_amt - (this_claim_amt_ex * 0.10)) < 0.02),
    CONSTRAINT FK_lines_claim       FOREIGN KEY (claim_id) REFERENCES dbo.claim_headers (claim_id) ON DELETE CASCADE
);

-- Indexes

Phase B — Validation & Audit Tables
Every Azure Function execution, rule result, and claim status change is permanently recorded here. The audit_trail table has a DDL-level constraint preventing UPDATE and DELETE — a SOPA legal requirement.
🔒
Legal Immutability: The audit_trail table is append-only by DDL constraint. No application role may UPDATE or DELETE. Violations are logged to Azure Monitor. This is non-negotiable under SOPA — the audit trail must be tamper-proof.

-- Phase B — Validation & Audit Tables
-- Deploy after Phase A. All tables reference claim_headers.

CREATE TABLE dbo.validation_runs (
    run_id          VARCHAR(30)   NOT NULL,    -- AZF-VAL-YYYYMMDD-XXXXX
    claim_id        VARCHAR(30)   NOT NULL,
    triggered_at    DATETIME2(3)  NOT NULL,
    total_rules     TINYINT       DEFAULT 9  NOT NULL,
    passed          TINYINT       DEFAULT 0  NOT NULL,
    warned          TINYINT       DEFAULT 0  NOT NULL,
    blocked         TINYINT       DEFAULT 0  NOT NULL,
    advisory        TINYINT       DEFAULT 0  NOT NULL,
    overall_result  VARCHAR(10)   NOT NULL,
    runtime_ms      INT,
    CONSTRAINT PK_validation_runs   PRIMARY KEY (run_id),
    CONSTRAINT CK_val_result        CHECK       (overall_result IN ('PASS','WARN','BLOCK')),
    CONSTRAINT FK_val_claim         FOREIGN KEY (claim_id) REFERENCES dbo.claim_headers (claim_id)
);

CREATE TABLE dbo.validation_rule_results (
    result_id    INT           IDENTITY(1,1)  NOT NULL,
    run_id       VARCHAR(30)                    NOT NULL,
    rule_number  TINYINT                        NOT NULL,
    rule_name    VARCHAR(60)                    NOT NULL,
    result       VARCHAR(10)                    NOT NULL,
    detail_json  NVARCHAR(MAX),
    affected_rowsSMALLINT,
    CONSTRAINT PK_rule_results   PRIMARY KEY (result_id),
    CONSTRAINT UQ_run_rule       UNIQUE      (run_id, rule_number),
    CONSTRAINT CK_rule_number    CHECK       (rule_number BETWEEN 1 AND 9),
    CONSTRAINT CK_rule_result    CHECK       (result IN ('PASS','WARN','BLOCK','ADVISORY')),
    CONSTRAINT FK_rr_run         FOREIGN KEY (run_id) REFERENCES dbo.validation_runs (run_id)
);

-- ⚠️ IMMUTABLE TABLE — SOPA LEGAL REQUIREMENT
-- No UPDATE or DELETE permitted. Enforced at DB role level.
-- See: DENY UPDATE, DELETE ON dbo.audit_trail TO [gridsync_app_role]
CREATE TABLE dbo.audit_trail (
    audit_id    BIGINT        IDENTITY(1,1)         NOT NULL,
    claim_id    VARCHAR(30)                          NOT NULL,
    changed_at  DATETIME2(3)  DEFAULT GETUTCDATE()   NOT NULL,
    changed_by  NVARCHAR(100)                        NOT NULL,
    from_status VARCHAR(20),
    to_status   VARCHAR(20)                          NOT NULL,
    reason      NVARCHAR(500),
    ip_address  VARCHAR(45),
    session_id  NVARCHAR(100),
    CONSTRAINT PK_audit_trail PRIMARY KEY (audit_id),
    CONSTRAINT FK_audit_claim FOREIGN KEY (claim_id) REFERENCES dbo.claim_headers (claim_id)
);

-- Deny mutating permissions at the DB role level
DENY UPDATE, DELETE ON dbo.audit_trail TO [gridsync_app_role];
